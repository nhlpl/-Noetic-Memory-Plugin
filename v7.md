    def route(self, task_complexity: float, context_length: int, 
              budget_sensitivity: float = 0.5, latency_requirement: str = "balanced") -> ModelTier:
        """
        Select the optimal model based on task requirements.
        
        Args:
            task_complexity: 0-1 score of task difficulty
            context_length: number of tokens in context
            budget_sensitivity: 0-1, higher = more cost-conscious
            latency_requirement: "fast", "balanced", or "slow"
        """
        candidates = []
        for tier in self.tiers.values():
            if context_length > tier.max_context:
                continue
                
            # Calculate composite score
            capability_match = 1.0 - abs(tier.capability_score - task_complexity)
            cost_efficiency = 1.0 / (tier.input_cost_per_1m + 0.01) * budget_sensitivity
            latency_match = 1.0 if tier.latency_profile == latency_requirement else 0.5
            
            score = capability_match * (1 - budget_sensitivity) + cost_efficiency * budget_sensitivity
            score += latency_match * 0.2
            
            candidates.append((tier, score))
            
        if not candidates:
            return self.tiers["deepseek_chat"]
            
        # Select highest scoring model
        best_tier, _ = max(candidates, key=lambda x: x[1])
        return best_tier
    
    def estimate_cost(self, tier: ModelTier, input_tokens: int, output_tokens: int) -> float:

        stats = self.usage_stats[tier_name]

        stats["calls"] += 1

class OrchestrationTopology(Enum):
    PARALLEL = "parallel"
    SEQUENTIAL = "sequential"
    HIERARCHICAL = "hierarchical"
    DEBATE = "debate"

class AdaptiveOrchestrator:
    """
    AdaptOrch-inspired dynamic topology selection.
    Chooses optimal agent workflow based on task structure.
    """
    
    def __init__(self):
        self.topology_performance: Dict[OrchestrationTopology, List[float]] = defaultdict(list)
        
    def analyze_task(self, query: str, context: Dict) -> OrchestrationTopology:
        """Determine optimal topology based on task characteristics."""
        # Task decomposition analysis
        if self._has_independent_subtasks(query):
            return OrchestrationTopology.PARALLEL
        elif self._requires_sequential_reasoning(query):
            return OrchestrationTopology.SEQUENTIAL
        elif self._requires_consensus(query):
            return OrchestrationTopology.DEBATE
        else:
            return OrchestrationTopology.HIERARCHICAL
            
    def _has_independent_subtasks(self, query: str) -> bool:
        indicators = ["and", "also", "additionally", "multiple", "various", "separate"]
        return any(ind in query.lower() for ind in indicators)
    
    def _requires_sequential_reasoning(self, query: str) -> bool:
        indicators = ["then", "after", "next", "finally", "step", "process", "workflow"]
        return any(ind in query.lower() for ind in indicators)
    
    def _requires_consensus(self, query: str) -> bool:
        indicators = ["best", "optimal", "should", "recommend", "decision", "choose"]
        return any(ind in query.lower() for ind in indicators)
    
    def record_performance(self, topology: OrchestrationTopology, latency: float, accuracy: float):
        score = accuracy / (latency + 0.1)
        self.topology_performance[topology].append(score)


class ChimeraPredictiveScheduler:
    """
    Chimera-inspired predictive scheduling for heterogeneous agents.
    Predicts latency and schedules to jointly optimize speed and accuracy.
    """
    
    def __init__(self):
        self.agent_profiles: Dict[str, Dict] = {}
        self.latency_predictor = self._build_predictor()
        
    def _build_predictor(self):
        # Simulated ML model for latency prediction
        return lambda agent_id, task_size: task_size * 0.01 + hash(agent_id) % 10 * 0.1
    
    def register_agent(self, agent_id: str, hardware: str, avg_latency_ms: float):
        self.agent_profiles[agent_id] = {
            "hardware": hardware,
            "avg_latency": avg_latency_ms,
            "load": 0.0,
            "success_rate": 1.0
        }
        
    def schedule(self, tasks: List[Dict], agents: List[str]) -> List[Tuple[str, str]]:
        """
        Assign tasks to agents to minimize overall latency while maintaining accuracy.
        Returns list of (task_id, agent_id) assignments.
        """
        assignments = []
        agent_loads = {aid: 0.0 for aid in agents}
        
        for task in tasks:
            # Predict latency for each agent
            predictions = []
            for aid in agents:
                pred_latency = self.latency_predictor(aid, task.get("size", 100))
                load_penalty = agent_loads[aid] * 1.5
                total_latency = pred_latency + load_penalty
                
                # Adjust for success rate
                success_rate = self.agent_profiles[aid]["success_rate"]
                effective_latency = total_latency / success_rate
                
                predictions.append((aid, effective_latency))
                
            # Select best agent
            best_agent, _ = min(predictions, key=lambda x: x[1])
            assignments.append((task["id"], best_agent))
            agent_loads[best_agent] += task.get("size", 100) * 0.01
            
        return assignments

class TacnodeContextLake:
    """
    Tacnode-inspired shared, live semantic context layer.
    Enables agents to instantly share knowledge with millisecond latency.
    """
    
    def __init__(self):
        self.contexts: Dict[str, Dict] = {}  # context_id -> {data, version, timestamp}
        self.subscribers: Dict[str, List[str]] = defaultdict(list)  # agent_id -> [context_ids]
        self.version_counter: Dict[str, int] = defaultdict(int)
        self._lock = asyncio.Lock()
        
    async def publish(self, context_id: str, data: Any, publisher_id: str) -> int:
        """Publish new context version. All subscribers notified."""
        async with self._lock:
            self.version_counter[context_id] += 1
            version = self.version_counter[context_id]
            
            self.contexts[context_id] = {
                "data": data,
                "version": version,
                "publisher": publisher_id,
                "timestamp": time.time()
            }
            return version
            
    async def subscribe(self, agent_id: str, context_id: str):
        self.subscribers[agent_id].append(context_id)
        
    async def get(self, context_id: str, min_version: int = 0) -> Optional[Dict]:
        if context_id not in self.contexts:
            return None
        ctx = self.contexts[context_id]
        if ctx["version"] >= min_version:
            return ctx
        return None
        
    async def get_updates(self, agent_id: str) -> List[Dict]:
        """Get all new updates for subscribed contexts."""
        updates = []
        for ctx_id in self.subscribers[agent_id]:
            ctx = self.contexts.get(ctx_id)
            if ctx:
                updates.append(ctx)
        return updates


class LedgerStateStigmergy:
    """
    Ledger-state stigmergy: agents coordinate via shared immutable ledger.
    Provides ACID guarantees via distributed SQL backend.
    """
    
    def __init__(self, db_path: str = "./swarm_ledger.db"):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self._init_schema()
        self._pending_writes: List[Tuple[str, Dict]] = []
        
    def _init_schema(self):
        cursor = self.conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS ledger (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                agent_id TEXT NOT NULL,
                operation TEXT NOT NULL,
                target TEXT,
                data TEXT,
                version INTEGER,
                timestamp REAL,
                signature TEXT,
                UNIQUE(agent_id, version)
            )
        """)
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS state (
                key TEXT PRIMARY KEY,
                value TEXT,
                version INTEGER,
                updated_at REAL
            )
        """)
        self.conn.commit()
        
    @contextmanager
    def transaction(self):
        """ACID transaction context."""
        try:
            yield self.conn
            self.conn.commit()
        except Exception:
            self.conn.rollback()
            raise
            
    def write(self, agent_id: str, operation: str, target: str, data: Dict) -> int:
        """Write to the immutable ledger."""
        cursor = self.conn.cursor()
        cursor.execute("SELECT MAX(version) FROM ledger WHERE agent_id = ?", (agent_id,))
        row = cursor.fetchone()
        version = (row[0] or 0) + 1
        
        timestamp = time.time()
        signature = hashlib.sha256(f"{agent_id}:{operation}:{target}:{version}:{timestamp}".encode()).hexdigest()
        
        with self.transaction() as conn:
            conn.execute("""
                INSERT INTO ledger (agent_id, operation, target, data, version, timestamp, signature)
                VALUES (?, ?, ?, ?, ?, ?, ?)
            """, (agent_id, operation, target, json.dumps(data), version, timestamp, signature))
            
            # Update state if operation modifies it
            if operation in ("SET", "UPDATE"):
                conn.execute("""
                    INSERT OR REPLACE INTO state (key, value, version, updated_at)
                    VALUES (?, ?, ?, ?)
                """, (target, json.dumps(data), version, timestamp))
                
        return version
        
    def read_state(self, key: str) -> Optional[Dict]:
        cursor = self.conn.cursor()
        cursor.execute("SELECT value FROM state WHERE key = ?", (key,))
        row = cursor.fetchone()
        if row:
            return json.loads(row[0])
        return None
        
    def get_ledger_entries(self, agent_id: str, since_version: int = 0) -> List[Dict]:
        cursor = self.conn.cursor()
        cursor.execute(
            "SELECT * FROM ledger WHERE agent_id = ? AND version > ? ORDER BY version",
            (agent_id, since_version)
        )
        columns = ["id", "agent_id", "operation", "target", "data", "version", "timestamp", "signature"]
        return [dict(zip(columns, row)) for row in cursor.fetchall()]

        ("Compare and contrast MSA, RLM, and ByteRover for long-context LLMs.", 0.5),
    ]
    
    for query, budget in queries:

        print(f"   Cost: ${response['cost_usd']:.6f} | Latency: {response['latency_ms']:.1f}ms")
        
    # Get final stats
    stats = swarm.get_stats()
    print(f"\n📊 Final Stats:\n{json.dumps(stats, indent=2)}")


if __name__ == "__main__":
    asyncio.run(main())

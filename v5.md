class CognitiveCompressor:
    """Structure-aware KV-cache compression with Φ-based token importance."""
    def __init__(self, max_tokens: int = 10000, compression_ratio: float = 0.3):
        self.max_tokens = max_tokens
        self.compression_ratio = compression_ratio
        self.cache: OrderedDict[str, Tuple[Any, float]] = OrderedDict()
        
    def compute_phi(self, tokens: List[str]) -> List[float]:
        """Compute Φ-based token importance (simplified)."""
        n = len(tokens)
        phi = np.ones(n)
        # Position weight (U-curve: important at start/end)
        for i in range(n):
            pos_weight = 1.0 + 0.5 * math.exp(-abs(i - n//2) / (n/4))
            phi[i] *= pos_weight
        # Information density (unique words)
        from collections import Counter
        counts = Counter(tokens)
        for i, token in enumerate(tokens):
            tf = counts[token] / n
            idf = math.log(1.0 / (1.0 + tf))
            phi[i] *= (tf * idf + 0.5)
        return phi.tolist()
        
    def compress(self, text: str) -> str:
        tokens = text.split()
        if len(tokens) <= self.max_tokens * self.compression_ratio:
            return text
        phi = self.compute_phi(tokens)
        keep_count = int(len(tokens) * self.compression_ratio)
        indices = np.argsort(phi)[-keep_count:]
        indices.sort()
        compressed = [tokens[i] for i in indices]
        return ' '.join(compressed)

class GoalGatedMemory:
    """Goal-conditioned gating and utility tagging for bounded memory."""
    def __init__(self, capacity: int = 100):
        self.capacity = capacity
        self.buffer: List[Dict] = []
        self.utility_scores: Dict[str, float] = {}
        
    def add(self, content: str, goal: str, importance: float = 0.5) -> bool:
        """Gate memory write based on relevance to current goal."""
        # Compute goal relevance (simplified)
        relevance = self._compute_relevance(content, goal)
        if relevance < 0.3:
            return False  # Gated out
            
        mem_id = hashlib.sha256(f"{content}:{time.time()}".encode()).hexdigest()[:12]
        utility = importance * relevance
        self.buffer.append({"id": mem_id, "content": content, "utility": utility})
        self.utility_scores[mem_id] = utility
        
        if len(self.buffer) > self.capacity:
            self._prune_lowest_utility()
        return True
        
    def _compute_relevance(self, content: str, goal: str) -> float:
        goal_words = set(goal.lower().split())
        content_words = set(content.lower().split())
        overlap = len(goal_words & content_words)
        return min(1.0, overlap / max(1, len(goal_words)))
        
    def _prune_lowest_utility(self):
        if not self.buffer:
            return
        self.buffer.sort(key=lambda x: x["utility"])
        removed = self.buffer.pop(0)
        if removed["id"] in self.utility_scores:
            del self.utility_scores[removed["id"]]
            
    def get_relevant(self, goal: str, limit: int = 5) -> List[str]:
        scored = []
        for item in self.buffer:
            relevance = self._compute_relevance(item["content"], goal)
            score = item["utility"] * relevance
            scored.append((score, item["content"]))
        scored.sort(key=lambda x: x[0], reverse=True)
        return [content for _, content in scored[:limit]]
